463 [Server] void CalculateScore()
464 {
465 int winningTeam = winningBidderIndex % 2;
466 int level = int.Parse(currentHighestBid.Substring(0, 1));
467 int requiredTricks = level + 6;
468 int tricksTaken = 0;
469 string suit = currentHighestBid.Substring(1).ToUpper();
470 for (int i = 0; i < 4; i++)
471 if (i % 2 == winningTeam) tricksTaken += tricksWonByPlayer[i];
472
473 if (tricksTaken >= requiredTricks)
474 {
475 int score = suit switch
476 {
477 "C" or "D" => 20 * level,
478 "H" or "S" => 30 * level,
479 "NT" => 40 + 30 * (level - 1),
480 _ => 0
481 };
482 pointsBelowLine[winningTeam] += score;
483 int overtricks = tricksTaken - requiredTricks;
484 int overtrickPoints = (suit == "C" || suit == "D") ? 20 * overtricks : 30 * overtricks;
485 pointsAboveLine[winningTeam] += overtrickPoints;
486 if (level == 6) pointsAboveLine[winningTeam] += 500;
487 if (level == 7) pointsAboveLine[winningTeam] += 1000;
488 }
489 else
490 {
491 int down = requiredTricks - tricksTaken;
492 int penalty = down * 50;
493 pointsAboveLine[1 - winningTeam] += penalty;
494 }
495
496 RpcUpdateScoreUI();
497
498 if (pointsBelowLine[winningTeam] >= 100)
499 {
500 partsWon[winningTeam]++;
501 if (partsWon[winningTeam] >= 2)
502 {
503 points[0] = pointsBelowLine[0] + pointsAboveLine[0];
504 points[1] = pointsBelowLine[1] + pointsAboveLine[1];
505 points[winningTeam] += partsWon[1 - winningTeam] == 0 ? 750 : 500;
506 int winner = points[0] > points[1] ? 0 : 1;
507 EndGame(winner);
508 return;
509 }
510 pointsAboveLine[0] += pointsBelowLine[0];
511 pointsAboveLine[1] += pointsBelowLine[1];
512 pointsBelowLine[0] = 0;
513 pointsBelowLine[1] = 0;
514 }
515 EndPlayPhase();
516 }
