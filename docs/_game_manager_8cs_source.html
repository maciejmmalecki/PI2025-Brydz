691 }
692
693 if (tricksTaken >= requiredTricks)
694 {
695 int score = 0;
696
697 switch (suit)
698 {
699 case "C":
700 case "D":
701 score = 20 * level;
702 break;
703 case "H":
704 case "S":
705 score = 30 * level;
706 break;
707 case "NT":
708 score = 40 + 30 * (level - 1);
709 break;
710 }
711
712 pointsBelowLine[winningTeam]+=score;
713
714 int overtricks= tricksTaken- requiredTricks;
715 int overtrickPoints=0;
716
717 switch (suit)

719 case "C":

722 break;
723 case "H":
724 case "S":
725 case "NT":
726 overtrickPoints = 30 * overtricks;
727 break;
728 }
729 pointsAboveLine[winningTeam]+=overtrickPoints;
730
731 if(level == 6){
732 pointsAboveLine[winningTeam]+=500;
733 }
734 if(level == 7){
735 pointsAboveLine[winningTeam]+=1000;
736 }
737 Debug.Log($"Zespół {winningTeam} zdobywa {score} punktów za kontrakt i {overtrickPoints} za dodatkowe lewy!");
738 }
739 else
740 {
741 int down = requiredTricks - tricksTaken;
742 int penalty = down * 50;
743 pointsAboveLine[1-winningTeam]+=penalty;
744 Debug.Log($"Zespół {1 - winningTeam} zdobywa {penalty} punktów za niewykonanie kontraktu.");
745 }
746 UpdateScoreUI();
747 if (pointsBelowLine[winningTeam] >= 100)
748 {
749 partsWon[winningTeam]++;
750
751 if (partsWon[winningTeam] >= 2)
752 {
753 points[0]=pointsAboveLine[0]+pointsBelowLine[0];
754 points[1]=pointsAboveLine[1]+pointsBelowLine[1];
755 if(partsWon[1-winningTeam]==0){
756 points[winningTeam]+=750;
757 }else{
758 points[winningTeam]+=500;
759 }
760 if(points[1-winningTeam]>points[winningTeam]){
761 Debug.Log($"Zespół {1-winningTeam} wygrywa cały mecz z wynikiem {(points[1-winningTeam]-points[winningTeam])/100}");

766 }

768 }

774 }
